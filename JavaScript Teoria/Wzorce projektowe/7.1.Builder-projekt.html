Dynamiczny Formularz

Kod HTML i JavaScript:


        <h1>Dynamiczny Formularz</h1>
        <div id="form-container"></div>

        <script>
        class Form {
            constructor() {
                this.fields = [];
            }

            addField(field) {
                this.fields.push(field);
                return this;
            }

            build() {
                return this.fields;
            }
        }

        const formBuilder = new Form();
        const myForm = formBuilder
            .addField({ label: "Imię", type: "text", value: "" })
            .addField({ label: "Email", type: "email", value: "" })
            .build();

        // Funkcja do dynamicznego renderowania formularza na stronie
        function renderForm(fields) {
            const formContainer = document.getElementById('form-container');
            const formElement = document.createElement('form');

            fields.forEach(field => {
                const label = document.createElement('label');
                label.textContent = field.label;

                const input = document.createElement('input');
                input.type = field.type;
                input.value = field.value;

                formElement.appendChild(label);
                formElement.appendChild(input);
                formElement.appendChild(document.createElement('br'));
            });

            formContainer.appendChild(formElement);
        }

        // Wyświetlamy formularz
        renderForm(myForm);
