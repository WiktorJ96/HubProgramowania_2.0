Menedżer Zadań - Wzorzec Prototype

Kod HTML i JavaScript:


        <h1>Menedżer Zadań - Wzorzec Prototype</h1>
        <button id="addTaskBtn">Dodaj zadanie</button>
        <div id="task-container"></div>

        <script>
// Prototyp zadania
const TaskPrototype = {
    title: "Zadanie domyślne",  // Domyślny tytuł zadania
    description: "Opis domyślnego zadania",  // Domyślny opis zadania
    priority: "normalne",  // Domyślny priorytet zadania

    // Metoda klonująca zadanie
    clone: function() {
        return Object.create(this);  // Klonowanie prototypu
    },

    // Metoda ustawiająca szczegóły zadania
    setTaskDetails: function(title, description, priority) {
        this.title = title;  // Ustawienie tytułu zadania
        this.description = description;  // Ustawienie opisu zadania
        this.priority = priority;  // Ustawienie priorytetu zadania
    },

    // Metoda wyświetlająca szczegóły zadania
    displayTask: function() {
        return `Zadanie: ${this.title}, Opis: ${this.description}, Priorytet: ${this.priority}`;  // Zwracanie szczegółów zadania
    }
};

// Obsługa przycisku dodającego zadanie
document.getElementById('addTaskBtn').addEventListener('click', () => {
    const newTask = TaskPrototype.clone();  // Klonowanie zadania z prototypu

    // Pobieranie szczegółów zadania od użytkownika
    const taskTitle = prompt('Podaj tytuł zadania: ', 'Nowe zadanie');
    const taskDescription = prompt('Podaj opis zadania: ', 'Opis zadania');
    const taskPriority = prompt('Podaj priorytet zadania (niski, normalne, wysoki): ', 'normalne');
    
    // Ustawianie szczegółów zadania
    newTask.setTaskDetails(taskTitle, taskDescription, taskPriority); 

    // Wyświetlanie zadania na stronie
    const taskContainer = document.getElementById('task-container');
    const taskElement = document.createElement('p');  // Tworzenie elementu <p> dla zadania
    taskElement.textContent = newTask.displayTask();  // Ustawianie tekstu zadania w elemencie
    taskContainer.appendChild(taskElement);  // Dodanie zadania do kontenera
});

        </script>
