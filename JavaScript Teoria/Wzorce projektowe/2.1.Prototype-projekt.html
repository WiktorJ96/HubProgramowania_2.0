Menedżer Zadań - Wzorzec Prototype

Kod HTML i JavaScript:


        <h1>Menedżer Zadań - Wzorzec Prototype</h1>
        <button id="addTaskBtn">Dodaj zadanie</button>
        <div id="task-container"></div>

        <script>
        const TaskPrototype = {
            title: "Zadanie domyślne",
            description: "Opis domyślnego zadania",
            priority: "normalne",
            clone: function() {
                return Object.create(this); // Klonowanie prototypu
            },
            setTaskDetails: function(title, description, priority) {
                this.title = title;
                this.description = description;
                this.priority = priority;
            },
            displayTask: function() {
                return `Zadanie: ${this.title}, Opis: ${this.description}, Priorytet: ${this.priority}`;
            }
        };

        document.getElementById('addTaskBtn').addEventListener('click', () => {
            const newTask = TaskPrototype.clone(); // Klonowanie zadania
            const taskTitle = prompt('Podaj tytuł zadania: ', 'Nowe zadanie');
            const taskDescription = prompt('Podaj opis zadania: ', 'Opis zadania');
            const taskPriority = prompt('Podaj priorytet zadania (niski, normalne, wysoki): ', 'normalne');
            
            newTask.setTaskDetails(taskTitle, taskDescription, taskPriority); // Ustawianie szczegółów zadania

            const taskContainer = document.getElementById('task-container');
            const taskElement = document.createElement('p');
            taskElement.textContent = newTask.displayTask();
            taskContainer.appendChild(taskElement);
        });
        </script>
