        <h1>System Zarządzania Użytkownikami - Wzorzec Konstruktor</h1>
        <form id="userForm">
            <label for="firstName">Imię:</label>
            <input type="text" id="firstName" required>
            
            <label for="lastName">Nazwisko:</label>
            <input type="text" id="lastName" required>
            
            <label for="role">Rola:</label>
            <select id="role">
                <option value="Administrator">Administrator</option>
                <option value="Użytkownik">Użytkownik</option>
            </select>
            
            <button type="submit">Dodaj użytkownika</button>
        </form>

        <h2>Lista Użytkowników:</h2>
        <div id="user-container"></div>

        <script>
// Konstruktor User
function User(firstName, lastName, role) {
    this.firstName = firstName;  // Imię użytkownika
    this.lastName = lastName;  // Nazwisko użytkownika
    this.role = role;  // Rola użytkownika

    // Metoda wyświetlająca informacje o użytkowniku
    this.displayInfo = function() {
        return `Użytkownik: ${this.firstName} ${this.lastName}, Rola: ${this.role}`;  // Zwraca informacje o użytkowniku
    };

    // Metoda do zmiany roli użytkownika
    this.changeRole = function(newRole) {
        this.role = newRole;  // Zmienia rolę użytkownika
    };
}

// Obsługa formularza tworzenia nowego użytkownika
document.getElementById('userForm').addEventListener('submit', function(event) {
    event.preventDefault();  // Zatrzymanie domyślnego zachowania formularza

    // Pobieranie danych z formularza
    const firstName = document.getElementById('firstName').value;  // Pobieranie imienia
    const lastName = document.getElementById('lastName').value;  // Pobieranie nazwiska
    const role = document.getElementById('role').value;  // Pobieranie roli

    // Tworzenie nowego użytkownika
    const newUser = new User(firstName, lastName, role); 

    // Wyświetlanie informacji o użytkowniku na stronie
    const userContainer = document.getElementById('user-container');
    const userElement = document.createElement('p');
    userElement.textContent = newUser.displayInfo();  // Wyświetlanie informacji o nowo utworzonym użytkowniku
    userContainer.appendChild(userElement);  // Dodanie elementu do kontenera na stronie
});

        </script>
