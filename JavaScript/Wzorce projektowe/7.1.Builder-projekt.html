Dynamiczny Formularz

Kod HTML i JavaScript:


        <h1>Dynamiczny Formularz</h1>
        <div id="form-container"></div>

        <script>
// Klasa reprezentująca formularz
class Form {
    constructor() {
        this.fields = [];  // Inicjalizacja pustej tablicy na pola formularza
    }

    // Metoda dodająca pole do formularza
    addField(field) {
        this.fields.push(field);  // Dodaje pole do tablicy pól formularza
        return this;  // Zwraca samego siebie, aby umożliwić wywołania łańcuchowe
    }

    // Metoda zwracająca zbudowany formularz (tablica pól)
    build() {
        return this.fields;  // Zwraca tablicę pól formularza
    }
}

// Użycie FormBuildera do tworzenia formularza
const formBuilder = new Form();  // Tworzenie nowej instancji formularza
const myForm = formBuilder
    .addField({ label: "Imię", type: "text", value: "" })  // Dodanie pola typu "text" dla imienia
    .addField({ label: "Email", type: "email", value: "" })  // Dodanie pola typu "email" dla adresu email
    .build();  // Budowanie formularza (zwraca gotową tablicę pól)

// Funkcja do dynamicznego renderowania formularza na stronie
function renderForm(fields) {
    const formContainer = document.getElementById('form-container');  // Pobieranie kontenera na formularz z DOM
    const formElement = document.createElement('form');  // Tworzenie elementu formularza

    // Iteracja po polach formularza i dynamiczne tworzenie etykiet i pól
    fields.forEach(field => {
        const label = document.createElement('label');  // Tworzenie elementu etykiety
        label.textContent = field.label;  // Ustawianie tekstu etykiety

        const input = document.createElement('input');  // Tworzenie elementu input
        input.type = field.type;  // Ustawianie typu input
        input.value = field.value;  // Ustawianie wartości input

        // Dodawanie etykiety i pola input do formularza
        formElement.appendChild(label);
        formElement.appendChild(input);
        formElement.appendChild(document.createElement('br'));  // Dodawanie elementu przerwy między polami
    });

    // Dodanie zbudowanego formularza do kontenera
    formContainer.appendChild(formElement);
}

// Wyświetlenie formularza na stronie
renderForm(myForm);  // Przekazanie pól formularza do funkcji renderującej
