Poniżej znajdują się fragmenty kodów SASS i HTML, z których składa się projekt:

Kod SASS


// Funkcja przyciemnienia koloru z ostrzeżeniem
@function safe-darken($color, $amount) {
    @if $amount > 20% {
        @warn "Uwaga: Przyciemnianie o więcej niż 20% może zmienić wygląd elementu";
    }
    @return darken($color, $amount);
}

// Funkcja ustawiająca szerokość z walidacją
@function set-width($width) {
    @if type-of($width) != 'number' {
        @error "Błąd: $width nie jest liczbą. Użyj wartości liczbowej.";
    } @else if $width > 100px {
        @warn "Ostrzeżenie: Szerokość większa niż 100px może wpłynąć na layout.";
    }
    @return $width;
}

// Debugowanie wartości
$font-size: 16px;
$line-height: 1.5;

@debug $font-size;
@debug $font-size * $line-height;

// Stylizacja elementów z użyciem funkcji i debugowaniem
.button-dark {
    background-color: safe-darken(#3498db, 25%);
    color: #fff;
    padding: 10px 20px;
}

.main-container {
    width: set-width(120px);
}

.text {
    font-size: $font-size;
    line-height: $line-height;
}
