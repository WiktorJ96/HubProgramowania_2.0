Podgląd Kodów SASS i HTML

Poniżej znajdują się fragmenty kodów SASS i HTML, z których składa się projekt:

Kod SASS


// Funkcja do konwersji pikseli na rem
@function px-to-rem($px, $base-font-size: 16px) {
    @return $px / $base-font-size * 1rem;
}

// Funkcja do przyciemniania koloru
@function darken-color($color, $percentage) {
    @return darken($color, $percentage);
}

// Funkcja zwracająca mapę kolorów
@function color-palette() {
    @return (
        primary: #3498db,
        secondary: #2ecc71,
        danger: #e74c3c
    );
}

// Funkcja do obliczania odstępów
@function spacing($size) {
    @if $size == small {
        @return 8px;
    } @else if $size == medium {
        @return 16px;
    } @else if $size == large {
        @return 24px;
    } @else {
        @error "Niewłaściwa wartość dla $size: musi być small, medium lub large";
    }
}

// Użycie funkcji w stylach
.container {
    font-size: px-to-rem(18px);
}

.button-dark {
    background-color: darken-color(#3498db, 20%);
    color: #fff;
    padding: 10px 20px;
}

.alert {
    background-color: map-get(color-palette(), danger);
    color: #fff;
    padding: 10px;
    border-radius: 5px;
}

.spacing-small { margin-bottom: spacing(small); }
.spacing-medium { margin-bottom: spacing(medium); }
.spacing-large { margin-bottom: spacing(large); }
